// Levantamiento POP pickup — Google Apps Script v4
// Pega este código en Extensiones > Apps Script
// Después: Implementar → Administrar → ✏️ → Nueva versión → Implementar
// ══════════════════════════════════════════════════════════════════════
const SHEET_NAME = 'Respuestas';
const FOLDER_ID  = '1lEeX-U5l3syy_46KqzfTt-8eKRzPXnj2';
const HEADERS    = ['ID','Fecha','Local','Nombre','Estacionamientos','N° Est.',
                    'Gráfica','Retiro','Foto','Drive File ID','Drive URL'];

// ─ Retorna la carpeta Drive por ID ───────────────────────────────────────────
function getFolder() {
  var folder = DriveApp.getFolderById(FOLDER_ID);
  Logger.log('[GAS] Carpeta: ' + folder.getName() + ' (ID: ' + FOLDER_ID + ')');
  return folder;
}

// ─ Inicializa cabeceras si la hoja está vacía ────────────────────────────
function initSheet(sheet) {
  if (sheet.getLastRow() === 0) {
    // Sheet vacío: crear headers y formato
    sheet.appendRow(HEADERS);
    sheet.getRange(1, 1, 1, HEADERS.length)
         .setFontWeight('bold').setBackground('#0053e2').setFontColor('#ffffff');
    sheet.setFrozenRows(1);
    Logger.log('[GAS] Hoja inicializada con cabeceras.');
  } else {
    // Sheet existente: migrar columnas faltantes (ej: Drive File ID, Drive URL)
    var nCols       = Math.max(sheet.getLastColumn(), HEADERS.length);
    var currentHdrs = sheet.getRange(1, 1, 1, nCols).getValues()[0];
    var updated     = false;
    HEADERS.forEach(function(h, i) {
      if (String(currentHdrs[i] || '').trim() !== h) {
        sheet.getRange(1, i + 1).setValue(h);
        Logger.log('[GAS] Header migrado col ' + (i+1) + ': "' + (currentHdrs[i]||'') + '" → "' + h + '"');
        updated = true;
      }
    });
    if (updated) SpreadsheetApp.flush();
  }
}

// ─ POST: guarda en Sheet → sube imagen a Drive → actualiza fila ─────────────
function doPost(e) {
  Logger.log('[GAS] ── doPost iniciado ──────────────────────────────────');

  // 1. Parsear JSON
  var data;
  try {
    data = JSON.parse(e.postData.contents);
    Logger.log('[GAS] Payload: id=' + data.id + ' | local=' + data.local
              + ' | extraOnly=' + (data.extraOnly || false));
  } catch (parseErr) {
    Logger.log('[GAS] ERROR parse: ' + parseErr.message);
    return out({ ok: false, error: 'JSON inválido: ' + parseErr.message });
  }

  // ── Rama especial: sólo subir foto a Drive, sin tocar el Sheet ──────────────
  if (data.extraOnly) {
    if (!data.photoB64 || !data.photoName) {
      return out({ ok: false, error: 'extraOnly: faltan photoB64 / photoName' });
    }
    try {
      var folder = getFolder();
      var bytes  = Utilities.base64Decode(data.photoB64);
      var blob   = Utilities.newBlob(bytes, data.photoMime || 'image/jpeg', data.photoName);
      var file   = folder.createFile(blob);
      file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      var fid = file.getId();
      Logger.log('[GAS] ✅ extraOnly OK → ' + data.photoName + ' | ' + fid);
      return out({ ok: true, driveFileId: fid,
        driveUrl: 'https://drive.google.com/uc?export=download&id=' + fid });
    } catch (err) {
      Logger.log('[GAS] ❌ extraOnly error: ' + err.message);
      return out({ ok: false, error: err.message });
    }
  }

  // 2. Guardar en Sheet PRIMERO — el registro persiste aunque Drive falle
  var ss, sheet, sheetRow;
  try {
    ss    = SpreadsheetApp.getActiveSpreadsheet();
    sheet = ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME);
    initSheet(sheet);
    sheet.appendRow([
      String(data.id      || ''),
      String(data.fecha   || ''),
      String(data.local   || ''),
      String(data.nombre  || ''),
      String(data.estac   || ''),
      Number(data.numEst  || 0),
      String(data.grafica || ''),
      String(data.retiro  || ''),
      String(data.photoName || ''),
      '',   // Drive File ID — se rellena después de subir
      ''    // Drive URL    — se rellena después de subir
    ]);
    sheetRow = sheet.getLastRow();
    Logger.log('[GAS] ✅ Sheet: fila ' + sheetRow + ' guardada');
  } catch (sheetErr) {
    Logger.log('[GAS] ❌ ERROR Sheet: ' + sheetErr.message);
    return out({ ok: false, error: 'Fallo Sheet: ' + sheetErr.message });
  }

  // 3. Subir imagen a Google Drive (solo si el payload incluye los campos de foto)
  if (!data.photoB64 || !data.photoMime || !data.photoName) {
    Logger.log('[GAS] Sin foto en payload — fila guardada sin imagen Drive');
    return out({ ok: true });
  }

  var driveFileId, driveUrl;
  try {
    Logger.log('[GAS] Subiendo a Drive (' + data.photoB64.length + ' chars base64)...');
    var folder = getFolder();
    var bytes  = Utilities.base64Decode(data.photoB64);
    var blob   = Utilities.newBlob(bytes, data.photoMime, data.photoName);
    var file   = folder.createFile(blob);
    driveFileId = file.getId();
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    driveUrl = 'https://drive.google.com/uc?export=download&id=' + driveFileId;
    Logger.log('[GAS] ✅ Drive OK → ' + driveFileId);

    // 4. Actualizar la fila: Drive File ID en col 10, Drive URL en col 11
    sheet.getRange(sheetRow, 10).setValue(driveFileId);
    sheet.getRange(sheetRow, 11).setValue(driveUrl);
    SpreadsheetApp.flush();  // fuerza commit antes de que el Web App retorne
    Logger.log('[GAS] ✅ Fila ' + sheetRow + ' actualizada con Drive URL');
  } catch (driveErr) {
    Logger.log('[GAS] ❌ ERROR Drive: ' + driveErr.message);
    // El registro YA está en el Sheet — NO se elimina, se marca como drive-error
    try {
      sheet.getRange(sheetRow, 10).setValue('drive-error');
      sheet.getRange(sheetRow, 11).setValue('drive-error');
      SpreadsheetApp.flush();
    } catch(_) {}
    return out({ ok: true, driveError: driveErr.message });
  }

  Logger.log('[GAS] ── doPost completado OK ────────────────────');
  return out({ ok: true, driveFileId: driveFileId, driveUrl: driveUrl });
}

// ─ GET: ping / list / delete / deleteAll ─────────────────────────────────
function doGet(e) {
  try {
    var action = (e.parameter || {}).action || '';
    var ss     = SpreadsheetApp.getActiveSpreadsheet();
    var sheet  = ss.getSheetByName(SHEET_NAME);

    if (action === 'ping')
      return out({ ok: true, version: 4,
                   rows: sheet ? Math.max(0, sheet.getLastRow() - 1) : 0 });

    if (action === 'list') {
      if (!sheet || sheet.getLastRow() < 2) return out([]);
      var rows = sheet.getDataRange().getValues();
      var hdrs = rows[0];
      return out(rows.slice(1).map(function(r) {
        var o = {}; hdrs.forEach(function(h, i) { o[h] = r[i]; });
        return {
          id:          String(o['ID']               || ''),
          fecha:       String(o['Fecha']             || ''),
          local:       String(o['Local']             || ''),
          nombre:      String(o['Nombre']            || ''),
          estac:       String(o['Estacionamientos']  || ''),
          numEst:      Number(o['N° Est.']           || 0),
          grafica:     String(o['Gráfica']           || ''),
          retiro:      String(o['Retiro']            || ''),
          photoName:   String(o['Foto']              || ''),
          driveFileId: String(o['Drive File ID']     || ''),
          driveUrl:    String(o['Drive URL']         || ''),
        };
      }));
    }

    if (action === 'delete') {
      var id = String((e.parameter || {}).id || '');
      if (!id) return out({ ok: false, error: 'id requerido' });
      if (!sheet || sheet.getLastRow() < 2) return out({ ok: true, deleted: false });
      var vals   = sheet.getDataRange().getValues();
      var idIdx  = vals[0].indexOf('ID');
      var fidIdx = vals[0].indexOf('Drive File ID');
      for (var i = vals.length - 1; i >= 1; i--) {
        if (String(vals[i][idIdx]) === id) {
          if (fidIdx !== -1 && vals[i][fidIdx])
            try { DriveApp.getFileById(String(vals[i][fidIdx])).setTrashed(true); } catch(_) {}
          sheet.deleteRow(i + 1);
          return out({ ok: true, deleted: true });
        }
      }
      return out({ ok: true, deleted: false });
    }

    if (action === 'deleteAll') {
      if (!sheet || sheet.getLastRow() < 2) return out({ ok: true, deleted: 0 });
      var vals   = sheet.getDataRange().getValues();
      var fidIdx = vals[0].indexOf('Drive File ID');
      if (fidIdx !== -1)
        vals.slice(1).forEach(function(row) {
          if (row[fidIdx])
            try { DriveApp.getFileById(String(row[fidIdx])).setTrashed(true); } catch(_) {}
        });
      var count = sheet.getLastRow() - 1;
      sheet.deleteRows(2, count);
      return out({ ok: true, deleted: count });
    }

    return out({ ok: false, error: 'action desconocida: ' + action });
  } catch (err) {
    Logger.log('[GAS] doGet ERROR: ' + err.message);
    return out({ ok: false, error: err.message });
  }
}

function out(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}

// ══ LIMPIEZA — ejecutar desde el editor (sin deploy) ═══════════════════════
function limpiarHoja() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  if (!sheet) { Logger.log('Hoja no encontrada'); return; }
  var last = sheet.getLastRow();
  if (last < 2) { Logger.log('Ya está vacía'); return; }
  sheet.deleteRows(2, last - 1);
  Logger.log('Listo: ' + (last - 1) + ' filas eliminadas');
}

// ══ DIAGNÓSTICO — ejecutar para verificar permisos Drive + Sheet ═══════════
function diagnostico() {
  Logger.log('=== DIAGNÓSTICO GAS v4 ===');
  try {
    var folder = DriveApp.getFolderById(FOLDER_ID);
    Logger.log('✅ Carpeta Drive OK → ' + folder.getName());
  } catch(e) { Logger.log('❌ Carpeta Drive: ' + e.message); }
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    Logger.log('✅ Sheet OK → ' + ss.getName());
  } catch(e) { Logger.log('❌ Sheet: ' + e.message); }
  Logger.log('=== FIN DIAGNÓSTICO ===');
}
